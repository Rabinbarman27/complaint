CREATE TABLE feedback_complaint_data (
    id INT AUTO_INCREMENT PRIMARY KEY,
    Operation VARCHAR(50),
    Given_by VARCHAR(50),
    Date_of_submission DATE,
    Depatment_section VARCHAR(100),
    Incident_description TEXT,
    Main_Error_category VARCHAR(50),
    Sub_Error_categor VARCHAR(100),
    active_error VARCHAR(5),
    latent_error VARCHAR(5),
    cognitive_error VARCHAR(5),
    non_cognitive_error VARCHAR(5),
    Root_cause TEXT,
    avg_impact_score DECIMAL(5,2),
    avg_freq_score DECIMAL(5,2),
    immediate_correction TEXT,
    corrective_action TEXT,
    preventive_action TEXT,
    patient_consequences VARCHAR(5),
    Risk_discription1 TEXT,
    impact_score1 INT,
    freq_score1 INT,
    Risk_discription2 TEXT,
    impact_score2 INT,
    freq_score2 INT,
    Risk_discription3 TEXT,
    impact_score3 INT,
    freq_score3 INT,
    Risk_discription4 TEXT,
    impact_score4 INT,
    freq_score4 INT,
    Risk_discription5 TEXT,
    impact_score5 INT,
    freq_score5 INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
